#include <iostream>
using namespace std;
int n,dop,n1;
string mas1[10] = { "белого","белой","черного","черной","зелёного","зелёной","красного","красной","жёлтого","жёлтой" };
string mas2[12] = { "крысы","коровы","тигра","зайца","дракона","змеи","лошади","овцы","обезьяны","курицы","собаки","свиньи" };
int main()
{
	setlocale(LC_ALL, "Russian");
	cout << "Введите год" << endl;
	cin >> n;
	if (n % 12 <= 5 && n % 12 >= 9)
	{
		dop = 1;
	}
	else
	{
		dop = 0; 
	}
	if ((n % 10) % 2 == 1)
	{
		if (dop != 1)
		{
			n1 = (n % 10) - 1;
		}
		else
		{
			n1 = (n % 10);
		}
	}
	if ((n % 10) % 2 == 0)
	{
		if (dop != 0)
		{
			n1 = (n % 10) + 1;
		}
		else
		{
			n1 = (n % 10);
		}
	}

	cout <<"год " << mas1[n1] << " ";
	cout << mas2[(n + 8) % 12];
	
	return 0;
}
